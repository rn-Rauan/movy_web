\ddentity{Role} & \ddfield{id} & Inteiro, PK & Identificador do papel de acesso. \\
& \ddfield{name} & Enum, único & Nome do papel disponível no RBAC: \texttt{ADMIN} ou \texttt{DRIVER}. \\
& \ddfield{createdAt} & Data/hora & Data de criação do registro. \\
& \ddfield{updatedAt} & Data/hora & Data da última atualização. \\
\addlinespace

\ddentity{User} & \ddfield{id} & UUID, PK & Identificador global do usuário. \\
& \ddfield{name} & Texto & Nome completo do usuário. \\
& \ddfield{email} & Texto, único & Credencial de login e contato; validada por \textit{Value Object}. \\
& \ddfield{passwordHash} & Texto & Senha armazenada somente como hash Bcrypt. \\
& \ddfield{telephone} & Texto, opcional & Telefone para contato. \\
& \ddfield{status} & Enum & Estado do usuário: \texttt{ACTIVE} ou \texttt{INACTIVE}; usado para exclusão lógica. \\
& \ddfield{createdAt} / \ddfield{updatedAt} & Data/hora & Controle de criação e atualização. \\
\addlinespace

\ddentity{Organization} & \ddfield{id} & UUID, PK & Identificador da organização de transporte. \\
& \ddfield{name} & Texto & Nome público ou comercial da organização. \\
& \ddfield{cnpj} & Texto, único & CNPJ validado por dígitos verificadores. \\
& \ddfield{email} & Texto, único & E-mail institucional da organização. \\
& \ddfield{telephone} & Texto & Telefone institucional. \\
& \ddfield{address} & Texto & Endereço completo. \\
& \ddfield{slug} & Texto, único & Identificador seguro para URLs públicas. \\
& \ddfield{status} & Enum & Estado da organização; permite exclusão lógica. \\
& \ddfield{createdAt} / \ddfield{updatedAt} & Data/hora & Controle de criação e atualização. \\
\addlinespace

\ddentity{OrganizationMembership} & \ddfield{userId} & UUID, PK/FK & Usuário vinculado à organização. \\
& \ddfield{roleId} & Inteiro, PK/FK & Papel exercido pelo usuário na organização. \\
& \ddfield{organizationId} & UUID, PK/FK & Organização do vínculo; base do isolamento por tenant. \\
& \ddfield{assignedAt} & Data/hora & Data de atribuição do papel. \\
& \ddfield{removedAt} & Data/hora, nulo & Data de remoção do vínculo; nulo quando o vínculo está ativo. \\
\addlinespace

\ddentity{Driver} & \ddfield{userId} & UUID, PK/FK & Usuário que possui perfil de motorista. \\
& \ddfield{cnh} & Texto, único & Número da CNH, validado por \textit{Value Object}. \\
& \ddfield{cnhCategories} & Lista/enum & Categorias da CNH (A, B, C, D, E), normalizadas e sem duplicidade. \\
& \ddfield{cnhExpiresAt} & Data & Validade da CNH. \\
& \ddfield{status} & Enum & Estado do perfil: \texttt{ACTIVE}, \texttt{INACTIVE} ou \texttt{SUSPENDED}. \\
& \ddfield{createdAt} / \ddfield{updatedAt} & Data/hora & Controle de criação e atualização. \\
\addlinespace

\ddentity{Vehicle} & \ddfield{id} & UUID, PK & Identificador do veículo. \\
& \ddfield{plate} & Texto, único & Placa nos formatos nacional ou Mercosul, validada por \textit{Value Object}. \\
& \ddfield{model} & Texto & Modelo do veículo. \\
& \ddfield{type} & Enum & Tipo do veículo: \texttt{VAN}, \texttt{BUS}, \texttt{MINIBUS} ou \texttt{CAR}. \\
& \ddfield{maxCapacity} & Inteiro & Capacidade máxima de passageiros. \\
& \ddfield{status} & Enum & Estado do veículo: \texttt{ACTIVE} ou \texttt{INACTIVE}. \\
& \ddfield{organizationId} & UUID, FK & Organização proprietária; aplica o isolamento multi-tenant. \\
& \ddfield{createdAt} / \ddfield{updatedAt} & Data/hora & Controle de criação e atualização. \\
\addlinespace

\ddentity{TripTemplate} & \ddfield{id} & UUID, PK & Identificador do modelo recorrente de viagem. \\
& \ddfield{departurePoint} & Texto & Local de partida da rota. \\
& \ddfield{destination} & Texto & Destino da rota. \\
& \ddfield{departureTimeOfDay} & Hora & Horário recorrente de partida, armazenado em UTC. \\
& \ddfield{arrivalTimeOfDay} & Hora & Horário recorrente estimado de chegada, armazenado em UTC. \\
& \ddfield{priceOneWay} & Decimal & Preço para trajeto somente ida. \\
& \ddfield{priceReturn} & Decimal & Preço para trajeto somente volta. \\
& \ddfield{priceRoundTrip} & Decimal & Preço para ida e volta. \\
& \ddfield{isRecurring} & Booleano & Indica se o modelo gera instâncias automaticamente. \\
& \ddfield{frequency} & Lista/enum & Dias da semana em que a rota ocorre. \\
& \ddfield{shift} & Enum & Turno da viagem: manhã, tarde ou noite. \\
& \ddfield{stops} & Lista de texto & Paradas disponíveis; no MVP não há entidade própria para paradas. \\
& \ddfield{defaultCapacity} & Inteiro & Capacidade padrão propagada para novas instâncias. \\
& \ddfield{autoCancelEnabled} & Booleano & Habilita auto-cancelamento por baixa arrecadação. \\
& \ddfield{minRevenue} & Decimal & Receita mínima esperada para manter a viagem. \\
& \ddfield{autoCancelOffset} & Inteiro & Janela de antecedência para auto-cancelamento. \\
& \ddfield{status} & Enum & Estado do modelo: \texttt{ACTIVE} ou \texttt{INACTIVE}. \\
& \ddfield{organizationId} & UUID, FK & Organização proprietária do modelo. \\
& \ddfield{isPublic} & Booleano & Define se a rota pode aparecer nas listagens públicas. \\
& \ddfield{createdAt} / \ddfield{updatedAt} & Data/hora & Controle de criação e atualização. \\
\addlinespace

\ddentity{TripInstance} & \ddfield{id} & UUID, PK & Identificador da viagem datada. \\
& \ddfield{driverId} & UUID, FK, nulo & Motorista atribuído à viagem. \\
& \ddfield{vehicleId} & UUID, FK, nulo & Veículo atribuído à viagem. \\
& \ddfield{totalCapacity} & Inteiro & Capacidade efetiva da viagem. \\
& \ddfield{tripStatus} & Enum & Estado operacional: \texttt{DRAFT}, \texttt{SCHEDULED}, \texttt{CONFIRMED}, \texttt{IN\_PROGRESS}, \texttt{FINISHED} ou \texttt{CANCELED}. \\
& \ddfield{tripTemplateId} & UUID, FK & Modelo de rota que originou a instância. \\
& \ddfield{departureTime} & Data/hora & Data e horário absolutos de partida. \\
& \ddfield{arrivalEstimate} & Data/hora & Previsão absoluta de chegada. \\
& \ddfield{organizationId} & UUID, FK & Organização responsável pela viagem. \\
& \ddfield{createdAt} / \ddfield{updatedAt} & Data/hora & Controle de criação e atualização. \\
\addlinespace

\ddentity{Booking} & \ddfield{id} & UUID, PK & Identificador da inscrição do passageiro. \\
& \ddfield{enrollmentDate} & Data/hora & Momento em que a inscrição foi realizada. \\
& \ddfield{status} & Enum & Estado da inscrição: \texttt{ACTIVE} ou \texttt{INACTIVE}. \\
& \ddfield{presenceConfirmed} & Booleano & Indica se a presença foi confirmada no embarque. \\
& \ddfield{enrollmentType} & Enum & Tipo de trajeto: \texttt{ONE\_WAY}, \texttt{RETURN} ou \texttt{ROUND\_TRIP}. \\
& \ddfield{recordedPrice} & Decimal & Preço congelado no momento da inscrição. \\
& \ddfield{userId} & UUID, FK & Passageiro inscrito. \\
& \ddfield{tripInstanceId} & UUID, FK & Viagem associada à inscrição. \\
& \ddfield{boardingStop} & Texto & Parada de embarque escolhida pelo passageiro. \\
& \ddfield{alightingStop} & Texto & Parada de desembarque escolhida pelo passageiro. \\
& \ddfield{organizationId} & UUID, FK & Organização da viagem; reforça o isolamento multi-tenant. \\
& \ddfield{activeKey} & Texto, único parcial & Impede mais de uma inscrição ativa do mesmo usuário na mesma viagem. \\
& \ddfield{createdAt} / \ddfield{updatedAt} & Data/hora & Controle de criação e atualização. \\
\addlinespace

\ddentity{Payment} & \ddfield{id} & UUID, PK & Identificador do pagamento. \\
& \ddfield{organizationId} & UUID, FK & Organização responsável pela cobrança. \\
& \ddfield{method} & Enum & Método: \texttt{MONEY}, \texttt{PIX}, \texttt{CREDIT\_CARD} ou \texttt{DEBIT\_CARD}. \\
& \ddfield{amount} & Decimal & Valor cobrado pela inscrição. \\
& \ddfield{status} & Enum & Estado: \texttt{PENDING}, \texttt{COMPLETED} ou \texttt{FAILED}. \\
& \ddfield{enrollmentId} & UUID, FK único & Inscrição associada; relação 1:1 com \texttt{Booking}. \\
& \ddfield{createdAt} / \ddfield{updatedAt} & Data/hora & Controle de criação e atualização. \\
\addlinespace

\ddentity{Plan} & \ddfield{id} & Inteiro, PK & Identificador do plano. \\
& \ddfield{name} & Enum, único & Nome comercial: \texttt{FREE}, \texttt{BASIC}, \texttt{PRO} ou \texttt{PREMIUM}. \\
& \ddfield{price} & Decimal & Preço do plano. \\
& \ddfield{description} & Texto & Descrição exibida ao usuário. \\
& \ddfield{maxVehicles} & Inteiro & Limite de veículos permitidos. \\
& \ddfield{maxDrivers} & Inteiro & Limite de motoristas permitidos. \\
& \ddfield{maxMonthlyTrips} & Inteiro & Limite mensal de viagens. \\
& \ddfield{durationDays} & Inteiro & Duração da assinatura em dias. \\
& \ddfield{isActive} & Booleano & Indica se o plano pode ser contratado. \\
& \ddfield{createdAt} / \ddfield{updatedAt} & Data/hora & Controle de criação e atualização. \\
\addlinespace

\ddentity{Subscription} & \ddfield{id} & UUID, PK & Identificador da assinatura. \\
& \ddfield{organizationId} & UUID, FK & Organização assinante. \\
& \ddfield{planId} & Inteiro, FK & Plano contratado. \\
& \ddfield{status} & Enum & Estado: \texttt{ACTIVE}, \texttt{PAST\_DUE} ou \texttt{CANCELED}. \\
& \ddfield{startDate} & Data/hora & Início da vigência. \\
& \ddfield{expiresAt} & Data/hora & Data de expiração calculada a partir do plano. \\
& \ddfield{createdAt} / \ddfield{updatedAt} & Data/hora & Controle de criação e atualização. \\
\addlinespace

\ddentity{TripScheduling\newline Config} & \ddfield{id} & UUID, PK & Identificador da configuração de agendamento. \\
& \ddfield{organizationId} & UUID, FK único & Organização configurada; relação 1:0..1. \\
& \ddfield{daysAhead} & Inteiro & Quantos dias à frente gerar instâncias; intervalo permitido de 1 a 90. \\
& \ddfield{enabled} & Booleano & Liga ou desliga os jobs automáticos da organização. \\
& \ddfield{createdAt} / \ddfield{updatedAt} & Data/hora & Controle de criação e atualização. \\
\addlinespace

\ddentity{AuditLog} & \ddfield{id} & UUID, PK & Identificador do registro de auditoria. \\
& \ddfield{organizationId} & UUID, FK & Organização relacionada ao evento auditado. \\
& \ddfield{userId} & UUID, FK & Usuário que realizou a ação. \\
& \ddfield{action} & Texto & Descrição objetiva da ação realizada. \\
& \ddfield{timestamp} & Data/hora & Momento da ação. \\
\end{longtable}
}

\noindent\textit{Observação}: a entidade \texttt{AuditLog} está modelada no esquema para rastreabilidade, mas ainda não é persistida por casos de uso no MVP.
